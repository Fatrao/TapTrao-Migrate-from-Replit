## PART 1: FAVICON

The favicon is not showing the TapTrao logo. Fix it now.

### Step 1 â€” Generate favicon from logo

Run this script in the project:

```javascript
// scripts/generate-favicon.js
// Run with: node scripts/generate-favicon.js
const sharp = require('sharp')
const path = require('path')

async function generateFavicons() {
  const input = path.join(process.cwd(), 'public', 'logo.png')
  
  // 32x32 PNG favicon
  await sharp(input).resize(32, 32).toFile(path.join(process.cwd(), 'public', 'favicon-32x32.png'))
  
  // 16x16 PNG favicon  
  await sharp(input).resize(16, 16).toFile(path.join(process.cwd(), 'public', 'favicon-16x16.png'))
  
  // 180x180 Apple touch icon
  await sharp(input).resize(180, 180).toFile(path.join(process.cwd(), 'public', 'apple-touch-icon.png'))
  
  console.log('Favicons generated')
}

generateFavicons()
```

If sharp is not installed: `npm install sharp`
